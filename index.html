def take(generator, k):
    generator = itertools.islice(generator, k)
    sequence = np.array(list(generator))
    return sequence


def multiplicative_congruent_method(seed=48271, betta=48271, modulus=2**31):
    while True:
        seed = (betta * seed) % modulus
        yield seed / modulus
        
        
def normal(mean=0.0, variance=1.0, n_base=12):
    
    base_random_variable_generator = multiplicative_congruent_method()
    
    while True:
        bases = take(base_random_variable_generator, k=n_base)
        x = np.sqrt(12 / len(bases)) * (np.sum(bases) - len(bases) / 2)
        yield mean + np.sqrt(variance) * x
        
        
def lognormal(mean, variance):
    
    normal_generator = normal(mean, variance)
    
    for x in normal_generator:
        yield np.exp(x)
        
        
def exponential(rate):
    
    base_random_variable_generator = multiplicative_congruent_method()
    
    for x in base_random_variable_generator:
        yield -np.log(x) / rate
        
        
def get_normal_cdf(mean, variance):
    def f(x):
        return stats.norm.cdf(x, loc=mean, scale=np.sqrt(variance))
    return f
        
        
def get_lognormal_cdf(mean, variance):
    def f(x):
        return stats.lognorm.cdf(x, s=np.sqrt(variance), scale=np.exp(mean))
    return f
        
        
def get_exponential_cdf(rate):
    def f(x):
        return stats.expon.cdf(x, scale=1/rate)
    return f
        
        
def get_cdf(sequence):
    values, counts = np.unique(sequence, return_counts=True)
    counts = counts / len(sequence)
    cdf = np.cumsum(counts)
    return values, cdf

        
def ks_test(sequence, cdf, significance=0.05):
    values, sample_cdf = get_cdf(sequence)
    cdf = cdf(values)
    statistics = np.sqrt(len(sequence)) * np.abs(sample_cdf - cdf).max()
    y = kolmogi(significance)
    return statistics <= y


def pearson_chi2_test(sequence, cdf, n_bins=20, significance=0.05):
    
    hist, bin_edges = np.histogram(sequence, bins=n_bins)
    hist = hist / len(sequence)
    
    p = [cdf(edge_stop) - cdf(edge_start) for edge_stop, edge_start in zip(bin_edges[1:], bin_edges[:-1])]
    
    statistics = len(sequence) * ((hist - p) ** 2 / p).sum()
    y = chi2.ppf(1.0 - significance, df=len(hist) - 1)
    
    return statistics <= y


def analyze(generator, expectation, variance, cdf, n_values=1000, n_attempts_hypothesis=100):
    
    random_sequence = take(generator, n_values)
    
    expectation_sample = random_sequence.mean()
    variance_sample = random_sequence.var(ddof=1)

    print(f'expectation_sample: {expectation_sample}')
    print(f'expectation: {expectation}')
    print(f'variance_sample: {variance_sample}')
    print(f'variance: {variance}')

    sns.displot(x=random_sequence)
    plt.show()
    
    for test in (ks_test, pearson_chi2_test):
        rate_error = []

        for k_attempts in range(n_attempts_hypothesis // 10, n_attempts_hypothesis, n_attempts_hypothesis // 10):
            n_error_type_1 = 0
            for _ in range(k_attempts):
                random_sequence = take(generator, n_values)
                is_hypothesis_true = test(random_sequence, cdf)
                n_error_type_1 += (1 - is_hypothesis_true)
            rate_error += [n_error_type_1 / k_attempts]

        plt.plot(rate_error, label=test.__name__);
        
    plt.legend()
    plt.show()
